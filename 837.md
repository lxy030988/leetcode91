# 代码 js

```js
/**
 * @param {number} n
 * @param {number} k
 * @param {number} maxPts
 * @return {number}
 */
var new21Game = function (n, k, maxPts) {
  const dp = new Array(n + 1).fill(0)
  dp[0] = 1
  let windowSum = 0

  for (let i = 1; i < n + 1; i++) {
    if (i - maxPts - 1 >= 0) {
      windowSum -= dp[i - maxPts - 1]
    }
    if (i - 1 < k) {
      windowSum += dp[i - 1]
    }
    dp[i] = windowSum * (1 / maxPts)
  }

  let res = 0
  for (let i = k; i <= n; i++) {
    res += dp[i]
  }
  return res
}
```

# 复杂度分析

- 时间复杂度：O(n+k)
- 空间复杂度：O(n+k)
