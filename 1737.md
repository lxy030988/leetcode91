# 代码 js

```js
/**
 * @param {string} a
 * @param {string} b
 * @return {number}
 */
var minCharacters = function (a, b) {
  let arr1 = new Array(26).fill(0)
  let arr2 = new Array(26).fill(0)
  for (let i in a) {
    arr1[a.charCodeAt(i) - 97]++
  }
  for (let i in b) {
    arr2[b.charCodeAt(i) - 97]++
  }
  let an = a.length,
    bn = b.length,
    asum = 0,
    bsum = 0,
    res = Number.MAX_SAFE_INTEGER
  for (let i = 0; i < 25; i++) {
    // 前缀和计算
    asum += arr1[i]
    bsum += arr2[i]
    // 找最小值
    res = Math.min(
      res, //上一次结果
      an + bn - arr1[i] - arr2[i], //a 和 b 改成相同字符
      an - asum + bsum, //改a > b
      bn - bsum + asum //改 b > a
    )
  }
  // z只有 a和b相同
  return Math.min(res, an + bn - arr1[25] - arr2[25])
}
```

# 复杂度分析

- 时间复杂度：O(n)
- 空间复杂度：O(n)
